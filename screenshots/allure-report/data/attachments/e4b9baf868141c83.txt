pm.test("Status code is 200", function () {
  pm.response.to.have.status(200);
});

pm.test("Response has accessToken and tokenType", function () {
  const json = pm.response.json();
  pm.expect(json).to.have.property("accessToken");
  pm.expect(json).to.have.property("tokenType");
  pm.expect(json.accessToken).to.be.a("string");
  pm.expect(json.tokenType).to.be.a("string");
  pm.collectionVariables.set("accessToken", json.accessToken);
});

pm.test("Login response under 500ms", function () {
  pm.expect(pm.response.responseTime).to.be.below(500);
});
